def svp_risk_guard_node(state: AgentState):
    # Intercept structural step lists before node execution
    planned_actions = state["next_steps"]
    
    res = requests.post(
        "https://svp-api.onrender.com/v1/audit",
        json={"steps": planned_actions}
    ).json()
    
    if res.get("overall") == "BLOCKED":
        return {"status": "TERMINATED", "error": res}
        
    return {"status": "PROCEED"}
